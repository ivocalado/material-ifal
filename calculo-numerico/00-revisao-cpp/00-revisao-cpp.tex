#include <stdio.h>
int main(int argc, char** argv) {
        int mes, numDias;
        printf("Digite o mes \n"); scanf("%d", &mes);
        switch (mes)
        {
                case 2: numDias = 29;
                break;
                case 4: case 6: case 9:
                case 11:
                        numDias = 30;
                break;
                default : numDias = 31;

}

\section{Vetores e matrizes}
\begin{frame}[fragile]{Vetores}
\begin{itemize}
 \item Um vetor é um conjunto de variáveis de um mesmo tipo de dado as quais são acessadas e referenciadas através da aposição de índices ao identificador do vetor
\end{itemize}

\begin{lstlisting}
int main(int a, char** b) {
        int v[10];
        v[0] = 10; //Modifica a primeira posição
        v[5] = 30; //Modifica a sexta posição
        return 0;
}
\end{lstlisting}

\end{frame}

\begin{frame}[fragile]{Cadeias de caracteres}
 \begin{itemize}
  \item Em C não existe um tipo primitivo capaz de armazenar uma cadeia de caracteres. Para tal, faz uso de vetores de caracteres
 \end{itemize}
 
 \begin{lstlisting}
int main(int a, char** b) {
        char s[10];
        printf("Digite um nome: ");
        scanf("%s", s);
        printf("Nome digitado: %s\n", s);
        return 0;
}  
 \end{lstlisting}


\end{frame}

\begin{frame}[fragile]{Matrizes}
\begin{itemize}
 \item Matrizes são estruturas indexadas utilizadas para representar certa quantidade de valores de um mesmo tipo
 \item Um vetor é um tipo especial de matriz, chamado de matriz \textit{unidimensional}
 \item Matrizes podem ter 1, 2, ..., n dimensões
\end{itemize}
\begin{lstlisting}
int main(int a, char** b) {
        int m[10][10];
        
        //Acessando a linha 0 e a coluna 1        
        m[0][1] = 10; 
        return 0;
}   
\end{lstlisting}


\end{frame}

\begin{frame}[fragile]{Ponteiros}

\begin{lstlisting}
#include <stdio.h>

int main(int a, char** b) {
        int f = 10;
        int *p = &f;
        printf("f = %d; *p = %d, p = %p\n", f, *p, p);
        return 0;
}
\end{lstlisting}

\begin{block}{}
 f = 10; *p = 10, p = 0x7fff56d17044
\end{block}

\end{frame}

\begin{frame}[fragile]{Implementando a função swap}

\begin{lstlisting}
#include <stdio.h>

void swap(int *a, int *b) {
        int k = *a;
        *a = *b;
        *b = k;
}

int main(int a, char** b) {
        int p1 = 10, p2 = 20;
        printf("p1 = %d; p2 = %d\n", p1, p2);

        swap(&p1, &p2);
        printf("p1 = %d; p2 = %d\n", p1, p2);
        return 0;
}

\end{lstlisting}

  \begin{block}{}
p1 = 10; p2 = 20
p1 = 20; p2 = 10

\end{block}

\end{frame}
