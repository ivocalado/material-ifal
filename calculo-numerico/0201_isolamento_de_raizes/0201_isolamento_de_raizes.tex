\begin{frame}
\begin{block}{Definição}
 Um número real $\xi$ é um \textbf{zero da função} $f(x)$ ou uma \textbf{raiz da equação} $f(x) = 0$ se $f(\xi) = 0$.
\end{block}

   \begin{block}{Etapas para o cálculo de raizes}
      \begin{itemize}
         \item[(i)] \textbf{Localização} ou \textbf{isolamento} das raízes, que consiste em obter um intervalo que contém a raiz; e
         \item[(ii)] \textbf{Refinamento}, que consiste em, escolhidas aproximações iniciais no intervalo encontrado no processo de localização, melhorá-las sucessivamente até se obter uma aproximação para a raiz dentro de uma precisão $\varepsilon$ prefixada.
      \end{itemize}
   \end{block}


 
\end{frame}

\section{Isolamento de Raízes}
\begin{frame}{Isolamento de Raízes}
    \begin{block}{Teorema de Bolzano}
      Se uma função contínua $f(x)$ assume valores de sinais opostos nos pontos extremos do intervalo $[a,b]$, isto é $f(a) \cdot f(b) < 0$, então o intervalo conterá, no mínimo, uma raiz da equação $f(x) = 0$, em outras palavras haverá, no mínimo, um número $\xi \in (a,b)$ tal que $f(\xi) = 0$.
   \end{block}
   \begin{figure}
      \begin{minipage}[c]{0.4 \linewidth}
         \centering\includegraphics[scale=0.25]{imagens/figura0301.jpg}
      \end{minipage} \hfill
      \begin{minipage}[c]{0.55 \linewidth}
         \centering\includegraphics[scale=0.25]{imagens/figura0302.jpg}
      \end{minipage}
   \end{figure}

\end{frame}

\begin{frame}
\begin{block}{}
      \textbf{Observação:} Sob as hipóteses do Teorema de Bolzano, se $f'(x)$ existir e preservar o sinal em $(a,b)$, então este intervalo contém um única equação de $f(x) = 0$.
\end{block}

   \begin{block}{Exemplo}
      Isolar os zeros reais de $f(x) = x^3 - 9x + 3$.
   \end{block}

   \textbf{Solução:} Construindo uma tabela de valores para $f(x)$ e considerando apenas os sinais temos:
      \begin{table}[ht]
         \begin{center}
            \footnotesize{\begin{tabular}{c|cccccccccccc}
               \hline
               $x$ & $-\infty$ & $-100$ & $-10$ & $-5$ & $-3$ & $-1$ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
               \hline
               $f(x)$ & $-$ & $-$ & $-$ & $-$ & $+$ & $+$ & $+$ & $-$ & $-$ & $+$ & $+$ & $+$ \\
               \hline
            \end{tabular}}
         \end{center}
      \end{table}

      Sabendo que $f(x)$ é contínua para qualquer $x$ real e observando as variações de sinal, podemos concluir que cada um dos intervalos $I_1 = [-5,-3]$, $I_2 = [0,1]$ e $I_3 = [2,3]$ contém pelo menos um zero de $f(x)$. $\blacksquare$
\end{frame}

\begin{frame}
    
    \begin{block}{Exercício}
      Isolar os zeros da função $f(x) = \sqrt{x} - 5e^{-x}$.
   \end{block}

   
   \textbf{Solução:} Temos que $D(f) = \mathbb{R}^+$ ($D(f) \equiv$ domínio de $f(x)$).

      Construindo uma tabela de valores com o sinal de $f(x)$ para determinados valores de $x$ temos
      \begin{table}[ht]
         \begin{center}
            \footnotesize{\begin{tabular}{c|ccccc}
               \hline
               $x$ & $0$ & $1$ & $2$ & $3$ & $\ldots$ \\
               \hline
               $f(x)$ & $-$ & $-$ & $+$ & $+$ & $\ldots$ \\
               \hline
            \end{tabular}}
         \end{center}
      \end{table}

      Analisando a tabela, vemos que $f(x)$ admite pelo menos um zero no intervalo $(1,2)$.

      Para saber se este zero é único neste intervalo devemos analisar o sinal de $f'(x)$:
      $$f'(x) = \frac{1}{2\sqrt{x}} + 5e^{-x} > 0, \,\forall x > 0$$

      Assim, podemos concluir que $f(x)$ admite um único zero em todo seu domínio de definição e este zero está no intervalo $(1,2)$. $\blacksquare$
   
\end{frame}

\begin{frame}
    Para analisar graficamente uma função ou equação é suficiente utilizar um dos seguintes processos:
   \begin{itemize}
      \item[(i)] esboçar o gráfico da função $f(x)$ e localizar as absisas dos pontos onde a curva intercepta o eixo $ox$;
      \item[(ii)] a partir da equação $f(x) = 0$, obter a equação equivalente $g(x) = h(x)$, esboçar os gráficos das funções $g(x)$ e $h(x)$ no mesmo eixo cartesiano e localizar os pontos $x$ onde as duas curvas se interceptam, pois neste caso $f(\xi) = 0 \Leftrightarrow g(\xi) = h(\xi)$
      \item[(iii)] usar os programas que traçam gráficos de funções, disponíveis em algumas calculadoras ou softwares matemáticos.
   \end{itemize}
\end{frame}

\section{Refinamento}
\begin{frame}
    \frametitle{Refinamento}
   \begin{block}{Definição}
      Um \textbf{método iterativo} consiste em uma sequência de instruções que são executadas passo a passo, algumas das quais são repetidas em ciclos.
   \end{block}

   A execução de um ciclo recebe o nome de \textbf{iteração}. Cada iteração utiliza resultados das iterações anteriores e efetua determinados testes que permitem verificar se foi atingido um resultado próximo o suficiente do resultado esperado.

\end{frame}

\begin{frame}
    \begin{figure}
      \centering\includegraphics[scale=0.62]{imagens/figura0210.jpg}
   \end{figure}
\end{frame}

\begin{frame}{Critérios de Parada}
    Existem duas interpretações para raiz aproximada que nem sempre levam ao mesmo resultado:

   \begin{block}{}
      $\overline{x}$ é raiz aproximada com precisão $\varepsilon$ se:
      \begin{itemize}
         \item[(i)] $|\overline{x} - \xi| < \varepsilon \hspace{0.5cm}$ ou
         \item[(ii)] $|f(\overline{x})| < \varepsilon$
      \end{itemize}
   \end{block}
   Para aplicar o teste $(i)$ sem conhecer $\xi$ podemos reduzir o intervalo que contém a raiz a cada iteração. Ao se conseguir um intervalo $[a,b]$ tal que:
   $$\xi \in [a,b] \hspace{0.5cm} \textrm{e} \hspace{0.5cm} b - a < \varepsilon$$
   então $\forall x \in [a,b],\; |x - \xi| < \varepsilon$. Portanto $\forall x \in [a,b]$ pode ser tomado como $\overline{x}$.
\end{frame}

\begin{frame}
   \textbf{Observação:} Nem sempre é possível ter as exigências $(i)$ e $(ii)$ satisfeitas simultaneamente. Os gráficos a seguir ilustram algumas possibilidades:
   \begin{figure}
      \centering\includegraphics[scale=0.28]{imagens/figura0212.jpg}
   \end{figure}
\end{frame}

\section{Mét. da Bissecção}
\begin{frame}
 \begin{center}
 \Huge Método da Bissecção
\end{center}
\end{frame}

\begin{frame}
   Seja a função $f(x)$ contínua no intervalo $[a,b]$ e tal que $f(a) \cdot f(b) < 0$ e supondo por simplicidade que o intervalor $(a,b)$ contenha uma única raiz da equação $f(x) = 0$.

   \begin{block}{Objetivo}
      Reduzir a amplitude do intervalo que contém a raiz até se atingir a precisão requerida: $b - a < \varepsilon$, usando para isto a sucessiva divisão de $[a,b]$ ao meio.
   \end{block}
   \begin{figure}
      \centering\includegraphics[scale=0.25]{imagens/figura0213.jpg}
   \end{figure}
\end{frame}

\begin{frame}
   \begin{block}{Exemplo}
      Encontre o zero da função $f(x) = x\,\log(x) - 1$ utilizando o Método da Bissecção.
   \end{block}

   \textbf{Solução: }Já vimos que a função $f(x)$ dada tem um zero no intervalo $(2,3)$.
      $$\footnotesize{x_0 = \displaystyle\frac{2 + 3}{2} = 2,5 \; \left\{ \begin{array}{l} f(2) = -0,3979 < 0 \\ f(3) = 0,4314 > 0 \\ f(2,5) = -0,00515 < 0 \end{array} \right. \; \Rightarrow \quad \left\{ \begin{array}{l} \xi \in (2,5;3) \\ a_1 = x_0 = 2,5 \\ b_1 = b_0 = 3 \end{array} \right.}$$
      $$\footnotesize{x_1 = \displaystyle\frac{2,5 + 3}{2} = 2,75 \; \left\{ \begin{array}{l} f(2,5) < 0 \\ f(3) > 0 \\ f(2,75) = 0,2082 > 0 \end{array} \right. \; \Rightarrow \quad \left\{ \begin{array}{l} \xi \in (2,5;2,75) \\ a_2 = a_1 = 2,5 \\ b_2 = x_1 = 2,75 \end{array} \right.}$$

      $ $

      Repetir o processo até atingir a precisão requerida.
   

\end{frame}

\begin{frame}
 \begin{figure}
      \centering\includegraphics[scale=0.39]{imagens/bisseccao.jpg}
   \end{figure}
\end{frame}

\begin{frame}
    Em alguma etapa do processo tem-se ou a raiz exata $\xi$ ou uma sequência infinita de intervalos encaixados $a_0$, $b_0$, $a_1$, $b_1$, $a_2$, $b_2$, $\ldots$, $a_n$, $b_n$, $\ldots$, tal que
   $$f(a_n) \cdots f(b_n) < 0 \hspace{0.25cm} n = 1, 2, 3, \ldots$$

   Como a cada iteração o intervalo $[a,b]$ é dividido o meio, na n-ésima iteração o comprimento do intervalo será:
   $$b_n - a_n = \frac{b_{n-1} - a_{n-1}}{2} = \frac{b_0 - a_0}{2^n}$$

   Deve-se obter o valor de $n$ tal que $b_n - a_n < \varepsilon$, ou seja,
   $$\frac{b_0 - a_0}{2^n} < \varepsilon \hspace{0.25cm} \Rightarrow \hspace{0.25cm} 2^n > \frac{b_0 - a_0}{\varepsilon} \hspace{0.25cm} \Rightarrow \hspace{0.25cm} \log(2^n) > \log\left(\frac{b_0 - a_0}{\varepsilon}\right)$$
   $$\Rightarrow \hspace{0.25cm} n\,\log(2) > \log(b_0 - a_0) - \log(\varepsilon) \hspace{0.25cm} \Rightarrow \hspace{0.25cm} n > \frac{\log(b_0 - a_0) - \log(\varepsilon)}{\log(2)}$$
\end{frame}

\begin{frame}
   \begin{block}{Exemplo}
      Se desejarmos encontrar $\xi$, o zero da função $f(x) = x\,\log(x) - 1$ que está no intervalo $[2,3]$ com precisão $\varepsilon = 10^{-2}$, quantas iterações, no mínimo, devemos efetuar?
   \end{block}

   \textbf{Solução:} Como $a_0 = 2$, $b_0 = 3$ e $\varepsilon = 10^{-2}$ temos que
      $$n = \frac{\log(3 - 2) - \log(10^{-2})}{\log(2)}$$
      $$n = \frac{\log(1) + 2\,\log(10)}{\log(2)}$$
      $$n = \frac{0 + 2}{0,3010}$$
      $$n \approx 6,64$$

      Logo, devemos efetuar $7$ iterações para atingir a precisão requerida.
\end{frame}

\begin{frame}
    \begin{block}{Exercício 1}

\section{Mét. da Posição Falsa}
\begin{frame}
 \begin{center}
 \Huge Método da Posição Falsa
\end{center}
\end{frame}

\begin{frame}
   Seja $f(x)$ uma função contínua em $[a,b]$, intervalo que contém uma raiz da equação $f(x) = 0$.

   Supor que o intervalo $(a,b)$ contenha uma única taiz da equação $f(x) = 0$. Podemos esperar conseguir a raiz aproximada $\overline{x}$ usando as informações sobre os valores de $f(x)$ disponíveis a cada iteração.
   
   No caso da bissecção, $x$ é simplesmente a média aritmética entre $a$ e $b$:
   
   $$x = \dfrac{a+b}{2}$$
   
   O objetivo do método da posição falsa é obter uma aproximação melhor que a média aritmética
   
   \pause
   \begin{block}{Intuição}
    Seja $f(x) = x^3-9x+3, [a,b] = [0,1]\,e\, f(1) = -5 < 0 < 3 = f(0)$
    
    $\xi$ ``parece'' estar mais próximo de $0$ que de $1$
   \end{block}

\end{frame}

\begin{frame}{Método da posição falsa}
 Em vez de tomarmos a média aritmética entre $a$ e $b$, tomamos a média aritmética ponderada, com pesos $|f(b)|$ e $|f(a)|$, respectivamente
 
 $$x = \dfrac{a|f(b)| + b|f(a)|}{|f(b)| + |f(a)|} = \dfrac{af(b) - bf(a)}{f(b) - f(a)}$$
 
 Graficamente, este ponto $x$ é a intersecção entre o eixo $\overrightarrow{ox}$ e a reta $r(x)$ que passa por $(a, f(a))$ e $bf(b)$
 
    \begin{figure}
      \centering\includegraphics[scale=0.24]{imagens/posicao-falsa.jpg}
   \end{figure}
 
\end{frame}


\begin{frame}
 O método da posição falsa aplicado a $xlog(x) - 1$ em $[a_0, b_0] = [2,3]$, fica:
 
 $$f(a_0) = -0,3979 < 0$$
 $$f(b_0) = 0,4314 > 0$$
 
 $$x_0 = \dfrac{a_0f(b_0) - b_0f(a_0)}{f(b_0) - f(a_0)} = \dfrac{2*0,4314 - 3*(-0,3979)}{0,4314 - (-0,3979)} = 2,4798$$
 
 $f(x_0) = -0,0219 < 0$. Como $f(a_0)$ e $f(x_0)$ têm o mesmo sinal,
 
 $a_1 = x_0 = 2,4798$
 
 $b_1 = b_0 = 3$
\end{frame}
