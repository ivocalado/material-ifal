\section{Definições}
\subsection{O que é Bootstrap?}
\begin{frame}{O que é Bootstrap?}
\begin{itemize}
 \item Bootstrap é um framework livre para construção de \textit{front-end} para desenvolvimento ágil e fácil de aplicações Web
 \item Inclue templates baseados em HTML e CSS para tipografia, formulários, botões, navegação entre outros
 \item Possibilita a criação de designs responsivos de maneira fácil
\end{itemize}
\pause
\begin{block}{O que é Design Web Responsivo?}
 Design Web Responsivo está relacionado a criação de web sites que automaticamente se ajustam para a apresentação adequada em todos os tipos de dispositivos, desde pequenos telefones até grandes desktops
\end{block}

\end{frame}

\begin{frame}{História do Bootstrap}
 \begin{itemize}
  \item O Bootstrap foi desenvolvido por \textbf{Mark Otto} e \textbf{Jacob Thornton} no Twitter e liberado como \textit{open source} em Agosto de 2011 no \textbf{GitHub}
 \end{itemize}
 \pause
 \begin{block}{Afinal, o que é mesmo o GitHub?}
 \pause
 Repositório de compartilhamento de projetos \textit{open source}! Permite que sejam criados projetos para armazenamento de código para desenvolvimento, download para público, wiki entre outras funcionalidades
 \end{block}


\end{frame}

\subsection{Vantagens}
\begin{frame}{Porque utilizar o Bootstrap?}
 Vantagens do Bootstrap:
 
 \begin{itemize}
  \item \textbf{Fácil de usar:} exige apenas o conhecimento básico de HTML e CSS para começar a usar o Bootstrap
  \item \textbf{Desenvolvimento responsivo:} todas as páginas que fazem uso do Bootstrap já se ajustam automaticamente a dispositivos com diferentes interfaces
  \item \textbf{\textit{Mobile first!}} todo o framework já é pensado em oferecer suporte à aplicações móveis
  \item \textbf{Compatibilidade com navegadores:} Bootstrap é compatível com todos os navegadores modernos (Chrome, Firefox, Internet Explorer, Safari, and Opera)
 \end{itemize}

\end{frame}

\begin{frame}{Como obter o Bootstrap?}
 Há duas formas de adicionar o Bootstrap à sua aplicação. Você pode: 
 
 \begin{enumerate}
  \item Fazer o download do Bootstrap de \url{http://getbootstrap.com}
  \item Usar o Bootstrap de um CDN (\textit{Content Delivery Network})
 \end{enumerate}

 \begin{block}{Porque escolher o CDN?}
  \begin{itemize}
   \item Não é necessário fazer o download
   \item O cliente pode reaproveitar o cache de outras páginas
   \item O CDN pode otimizar disponibilizando o servidor mais próximo do cliente
  \end{itemize}

 \end{block}

\end{frame}

\begin{frame}[fragile]{Como adicionar o Bootstrap à uma página}
 \begin{lstlisting}

<link rel="stylesheet" href="http://maxcdn.bootstrapcdn.com/bootstrap/3.3.4/css/bootstrap.min.css">

 \end{lstlisting}

\end{frame}

\begin{frame}[fragile]{Etapa 2}
O Bootstrap é é projetado para ser responsivo para dispositivos móveis. Para garantir a renderização apropriada e o zoom correto, adicione a seguinte tag meta dentro de head

\begin{lstlisting}
<meta name="viewport" content="width=device-width, initial-scale=1">
\end{lstlisting}

\begin{itemize}
 \item \textbf{width=device-width} ajusta a largura da página para seguir a largura da tela do dispositivo (que pode variar de dispositivo para dispositivo)
 
 \item \textbf{initial-scale=1} ajusta o zoom inicial quando a página é inicialmente carregada pelo navegador
\end{itemize}


\end{frame}

\begin{frame}[fragile]{Containers}
 Bootstrap requer a indicação do elemento mais externo a todos os conteúdos da página. Estão definidos duas classes \textit{containers} que podem ser escolhidas
 
 \begin{itemize}
  \item \textbf{.container} utiliza uma largura de \textit{container} fixa
  \item \textbf{.container-fluid} utiliza a largura total do device como largura do container
 \end{itemize}
 
 \begin{lstlisting}
<body>
    <div class="container">
      <h1>My First Bootstrap Page</h1>
      <p>This is some text.</p>
    </div>
<body>
 \end{lstlisting}

\end{frame}

 \item Bootstrap possibilita a criação de 7 estilos diferentes de links
 \begin{itemize}
  \item btn-default
    \item btn-primary
    \item btn-success
    \item btn-info
    \item btn-warning
    \item btn-danger
    \item btn-link

 \end{itemize}

\end{itemize}
\begin{figure}
 \centering
 \includegraphics[scale=0.45]{./imagens/buttons.png}
 % buttons.png: 0x0 pixel, 300dpi, 0.00x0.00 cm, bb=
\end{figure}

\begin{itemize}
 \item Para a criação dos elementos, deve-se adicionar a classe padrão de botões \textbf{btn}
 \item Estilos de botões podem ser adicionados a elementos \textbf{a}, \textbf{input} e \textbf{button}
\end{itemize}

\begin{lstlisting}
<a href="#" class="btn btn-info" role="button">Link Button</a>
<button type="button" class="btn btn-info">Button</button>
<input type="button" class="btn btn-info" value="Input Button">
<input type="submit" class="btn btn-info" value="Submit Button">
\end{lstlisting}

\begin{itemize}
 \item É possível configurar o tamanho dos botões a partir das seguintes classes:
 \begin{itemize}

\begin{frame}[fragile]{Collapse (ex-28)}
 \begin{itemize}
  \item Com o Bootstrap é possível esconder elementos inteiros HTML
 \end{itemize}
 
 \begin{lstlisting}
  <button data-toggle="collapse" data-target="#demo">Collapsible</button>

  <div id="demo" class="collapse">
    Lorem ipsum dolor text....
  </div>
 \end{lstlisting}
\begin{itemize}
 \item A classe \textbf{collapse} indica o elemento que será ocultado
 \item O elemento \textbf{data-target="\#id"} conecta o elemento que será ocultado do elemento ocultador
 \item Para tornar elemento ocultado ativo, basta adicionar a class \textbf{in}
\end{itemize}


\end{frame}

\subsection{Barras de Navegação}
\begin{frame}[fragile]{Barras de navegação}
\begin{itemize}
 \item Barras de navagação são cabeçalhos de navegação posicionados no topo da página
 \item Com o Bootstrap a barra de navegação pode ser expandida ou comprimida, dependendo do tamanho da tela
 \item Para criação de tabelas, devemos fazer uso da tag \textbf{nav} com as classes \textbf{navbar} e \textbf{navbar-default}
 
\end{itemize}
\begin{lstlisting}
<nav class="navbar navbar-default">
  <div class="container-fluid">
    <div>
      <ul class="nav navbar-nav">
        <li class="active"><a href="#">Home</a></li>
        <li><a href="#">Page 1</a></li>
      </ul></div>
  </div> </nav>
\end{lstlisting}

\end{frame}

\begin{frame}[fragile]{Barra de navegação}

\begin{lstlisting}
<nav class="navbar navbar-default">
  <div class="container-fluid">
    <div class="navbar-header">
      <a class="navbar-brand" href="#">WebSiteName</a>
    </div>
    <div>
      <ul class="nav navbar-nav">
        <li class="active"><a href="#">Home</a></li>
        <li><a href="#">Page 1</a></li>

        <li><a href="#">Page 3</a></li>
      </ul>

</nav>

\end{lstlisting}

 
\end{frame}

\begin{frame}[fragile]{Barra de naveração fixa}
\begin{itemize}
 \item É possível configurar a barra de navegação para que ela fique fixa no topo ou na baseados
 \item Basta utilizar a classe \textbf{navbar-fixed-top} ou \textbf{navbar-fixed-bottom}
 
\end{itemize}
\begin{lstlisting}
 <nav class="navbar navbar-inverse navbar-fixed-top">
  <div class="container-fluid">
      ....
\end{lstlisting}

\begin{block}{Como resolver o conteúdo sobrescrito pela barra de navegação?}
Basta adicionar a propriedade \textbf{padding-*: 70px} ao \textit{body}
 \begin{lstlisting}
<body style="padding-top: 70px;"> ou 
<body style="padding-bottom: 70px;">
 \end{lstlisting}

\end{block}
