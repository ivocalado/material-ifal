\section{Objetivos}
\begin{frame}{Objetivos}
\begin{block}{Objetivo Geral}
 Capacitar o aluno a resolver problemas através de soluções utilizando os conceitos de orientação a objetos. 
\end{block}
Específicos:
 
\begin{itemize}
  \item Compreender os conceitos fundamentais da linguagem utilizada;
  \item Compreender os conceitos de orientação a objetos;
  \item Implementar software utilizando o paradigma de orientação a objetos.
\end{itemize}
\end{frame}
 
\section{Ementa}
\begin{frame}[allowframebreaks]{Ementa}
Conteúdo básico:
 \begin{itemize}
  \item Introdução à OO \textbf{(1º Bimestre)}
  \begin{itemize}
   \item Motivação
   \item POO x Programação Estruturada
   \item Conceitos de classes, objetos, atributos e métodos
  \end{itemize}
  \item Fundamentos da linguagem de programação OO
  \begin{itemize}
   \item Introdução e Motivação da linguagem Java
   \item Primeiros passos no ambiente de programação
   \item Elementos básicos da linguagem e sua utilização (variáveis, operadores, comandos de entrada e saída, estrutura de controle, strings, arrays etc)
  \end{itemize}
  \item Fundamentos da POO
  \begin{itemize}
   \item Classes, objetos, atributos, métodos e construtores
   \item Encapsulamento, herança, polimorfismo, classes abstratas e interface  \textbf{(2º Bimestre)}
  \end{itemize}
  \item Estrutura de dados \textbf{(3º Bimestre)}
  \begin{itemize}
   \item Listas
   \item Filas
   \item Pilhas
   \item Mapas
  \end{itemize}
  \item Tratamento de exceções
  \item Programação com interface gráfica  
  \item Integração com Banco de Dados \textbf{(4º Bimestre)}
  \item Testes unitários
 \end{itemize}
\end{frame}
